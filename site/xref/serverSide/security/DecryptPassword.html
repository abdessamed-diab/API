17  /**
18   * decrypt an encrypted text.
19   * encrypt strategy is not provided for security reasons.
20   * note: developer can get original password value from memory after executing decrypt method.
21   */
22  public final class DecryptPassword {
23      private static final Logger LOGGER = Logger.getLogger(DecryptPassword.class.getSimpleName());
24  
25      public static String decrypt(String saltEncryptedPassword) throws BadPaddingException, IOException {
26          if (StringUtils.isBlank(saltEncryptedPassword)) {
27              LOGGER.warning("password should not be empty: "+saltEncryptedPassword);
28              throw new IllegalArgumentException("check your mail server password: "+saltEncryptedPassword);
29          }
30          byte[] salt = new String("12345678").getBytes();
31          int iterationCount = 40000;
32          int keyLength = 128;
33          SecretKeySpec key = createSecretKey("secret".toCharArray(), salt, iterationCount, keyLength);
34          return decrypt(saltEncryptedPassword, key);
35      }
36  
37      private static SecretKeySpec createSecretKey(char[] password, byte[] salt, int iterationCount, int keyLength) {
38          SecretKey keyTmp = null;
39          try {
40              SecretKeyFactory keyFactory = SecretKeyFactory.getInstance("PBKDF2WithHmacSHA512");
41              PBEKeySpec keySpec = new PBEKeySpec(password, salt, iterationCount, keyLength);
42              keyTmp = keyFactory.generateSecret(keySpec);
43          } catch (NoSuchAlgorithmException | InvalidKeySpecException e) {
44              e.printStackTrace();
45          }
46  
47          return new SecretKeySpec(keyTmp.getEncoded(), "AES");
48      }
49  
50  
51      private static String decrypt(String string, SecretKeySpec key)
52              throws BadPaddingException, IOException {
53          String iv = string.split(":")[0];
54          String property = string.split(":")[1];
55          try {
56              Cipher pbeCipher = Cipher.getInstance("AES/CBC/PKCS5Padding");
57              pbeCipher.init(Cipher.DECRYPT_MODE, key, new IvParameterSpec(base64Decode(iv)));
58              return new String(pbeCipher.doFinal(base64Decode(property)), "UTF-8");
59          } catch (NoSuchPaddingException | InvalidKeyException | InvalidAlgorithmParameterException | NoSuchAlgorithmException | IllegalBlockSizeException e) {
60              e.printStackTrace();
61          }
62  
63          return null;
64      }
65  
66      private static byte[] base64Decode(String property) throws IOException {
67          return Base64.getDecoder().decode(property);
68      }
